function selectAns(qIdx,val,btn){
  responses[currentDim][qIdx]=val;

function nextDim(){
  const unanswered=responses[currentDim].findIndex(r=>r===0);
  if(unanswered!==-1){
    const card=$(`qcard-${unanswered}`);

function calcScores(){
  return responses.map((res,di)=>{
    let sum=0;
    res.forEach((r,i)=>{sum+=DIMS[di].reversed.includes(i)?6-r:r;});
    return sum;
  });
}

function getClassification(iga){
  return PROFILES.find(p=>iga>=p.min&&iga<=p.max)||PROFILES[PROFILES.length-1];
}

function getLevelInfo(pct){
  if(pct>=85)return{label:'Excelente',color:'#f0c040'};<br/>
  if(pct>=70)return{label:'Elevado',color:'#27ae60'};<br/>

  if(pct>=40)return{label:'Baixo',color:'#e67e22'};<br/>
  return{label:'Cr√≠tico',color:'#e74c3c'};
}

function drawGauge(pct){
  const canvas=$('gaugeCanvas');
  const ctx=canvas.getContext('2d');
  const cx=140,cy=140,r=110,lw=18;
  ctx.clearRect(0,0,280,155);
  
  // Segmentos coloridos
  const segs=[
    {start:Math.PI,end:Math.PI*1.2,color:'#e74c3c'},<br/>
    {start:Math.PI*1.2,end:Math.PI*1.4,color:'#e67e22'},<br/>
    {start:Math.PI*1.4,end:Math.PI*1.6,color:'#f1c40f'},<br/>
    {start:Math.PI*1.6,end:Math.PI*1.8,color:'#27ae60'},<br/>
    {start:Math.PI*1.8,end:Math.PI*2,color:'#f0c040'}
  ];
  segs.forEach(s=>{
    ctx.beginPath();ctx.arc(cx,cy,r,s.start,s.end);
    ctx.lineWidth=lw;ctx.strokeStyle=s.color;
    ctx.lineCap='butt';ctx.stroke();
  });
  
  // Track fundo
  ctx.beginPath();ctx.arc(cx,cy,r,Math.PI,0);
  ctx.lineWidth=lw+2;ctx.strokeStyle='rgba(255,255,255,0.05)';ctx.stroke();
  segs.forEach(s=>{
    ctx.beginPath();ctx.arc(cx,cy,r,s.start,s.end);
    ctx.lineWidth=lw;ctx.strokeStyle=s.color;ctx.stroke();
  });
  
  // Agulha
  const angle=Math.PI+(pct/100)*Math.PI;
  const nx=cx+r*0.82*Math.cos(angle),ny=cy+r*0.82*Math.sin(angle);
  ctx.beginPath();ctx.moveTo(cx,cy);ctx.lineTo(nx,ny);
  ctx.lineWidth=3;ctx.strokeStyle='#fff';ctx.lineCap='round';ctx.stroke();
  ctx.beginPath();ctx.arc(cx,cy,8,0,Math.PI*2);ctx.fillStyle='#fff';ctx.fill();
  
  // Texto IGA
  ctx.font='bold 28px Poppins,Inter,sans-serif';
  ctx.fillStyle='#f0c040';ctx.textAlign='center';ctx.textBaseline='middle';
  ctx.fillText(pct.toFixed(1)+'%',cx,cy-20);
  ctx.font='12px Inter,sans-serif';
  ctx.fillStyle='rgba(255,255,255,0.6)';
  ctx.fillText('√çndice Global de Autoestima',cx,cy+5);
}
